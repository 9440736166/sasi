//jacascript code let slideIndex = 0; let slides = document.getElementsByClassName("mySlides"); function showSlides() { for (let i = 0; i < slides.length; i++) { slides[i].style.display = "none"; } slideIndex++; if (slideIndex > slides.length) { slideIndex = 1; } slides[slideIndex - 1].style.display = "block"; } function toggleSlides() { if (slides[0].style.display === "none") { for (let i = 0; i < slides.length; i++) { slides[i].style.display = "block"; } } else { for (let i = 0; i < slides.length; i++) { slides[i].style.display = "none"; } } } setInterval(showSlides, 2000); // Change slide every 2 seconds (2000 milliseconds)
